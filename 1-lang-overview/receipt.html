console.log("---------------------") console.log("PART 1...") function formatUSD(price) { return "$" + price.toFixed(2); } let formattedPrice = formatUSD(4.6777); console.log("Formatted USD:", formattedPrice); // Should log: "$4.68" console.log("---------------------") console.log("PART 2...") function formatPct(percentage) { return (percentage * 100).toFixed(2) + "%"; } let formattedPct = formatPct(0.0875); console.log("Formatted Percentage:", formattedPct); // Should log: "8.75%" console.log("---------------------") console.log("PART 3...") let subtotal = 19.41; let taxRate = 0.0875; let tax = subtotal * taxRate; let total = subtotal + tax; console.log("Subtotal:", formatUSD(subtotal)); // "$19.41" console.log("Tax Rate:", formatPct(taxRate)); // "8.75%" console.log("Tax Owed:", formatUSD(tax)); // "$1.70" console.log("Total:", formatUSD(total)); // "$21.11" console.log("---------------------") console.log("PART 4...") function generateReceipt(subtotal, taxRate = 0.0875) { let tax = subtotal * taxRate; let total = subtotal + tax; return { subtotal: subtotal, taxRate: taxRate, tax: tax, total: total }; } let receipt = generateReceipt(19.41); console.log("Subtotal:", formatUSD(receipt.subtotal)); console.log("Tax Rate:", formatPct(receipt.taxRate)); console.log("Tax Owed:", formatUSD(receipt.tax)); console.log("Total:", formatUSD(receipt.total));
